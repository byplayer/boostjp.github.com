#include <list>
#include <algorithm>
#include <functional>
#include <cstdlib>
#include "print.hpp"
#include "compose.hpp"
using namespace std;
using namespace boost;


int main()
{
    list<int> coll;

    // 5つの乱数を挿入する
    generate_n (back_inserter(coll),      // 挿入位置
                5,                        // 数
                rand);                    // 乱数生成オブジェクト
    PRINT_ELEMENTS(coll);

    // 5つの要素を0以上10未満の乱数で置き換える
    generate (coll.begin(), coll.end(),   // 置換の対象となる範囲
              compose_f_g(bind2nd(modulus<int>(),10),
                          ptr_fun(rand)));
    PRINT_ELEMENTS(coll);
}
