#include <iostream>
#include <algorithm>
#include <functional>
#include <string>
#include <cctype>
#include "compose.hpp"
using namespace std;
using namespace boost;

int main()
{
    string s("Internationalization");
    string sub("Nation");

    // 大文字/小文字を区別なしに部分文字列を検索する
    string::iterator pos;
    pos = search (s.begin(),s.end(),           // 検索対象文字列
                  sub.begin(),sub.end(),       // 検索する部分文字列
                  compose_f_gx_hy(equal_to<int>(), // 一致条件
                                  ptr_fun(::toupper),
                                  ptr_fun(::toupper)));

    if (pos != s.end()) {
        cout << "\"" << sub << "\" is part of \"" << s << "\""
             << endl;
    }
}
